## 🧠 CORE LOGIC: USE ATR (Average True Range)

Add ATR calculation (or approximate if not available):

```ts
function calculateATR(highs: number[], lows: number[], closes: number[]): number {
  const ranges = highs.map((h, i) => Math.max(
    h - lows[i],
    Math.abs(h - closes[i - 1] || h),
    Math.abs(lows[i] - closes[i - 1] || lows[i])
  ));
  return ranges.reduce((a, b) => a + b, 0) / ranges.length;
}
```

## 🔧 SL/TP GENERATION

```ts
function generateSLTP({
  entry,
  direction,
  atr,
  style
}: {
  entry: number;
  direction: "LONG" | "SHORT";
  atr: number;
  style: "scalp" | "day" | "swing";
}) {
  const config = styleConfig[style];

  let slDistance = atr * config.slMultiplier;
  let tpDistance = atr * config.tpMultiplier;

  // Clamp SL (important!)
  slDistance = Math.min(slDistance, config.maxSL);

  let stopLoss, takeProfit;

  if (direction === "LONG") {
    stopLoss = entry - slDistance;
    takeProfit = entry + tpDistance;
  } else {
    stopLoss = entry + slDistance;
    takeProfit = entry - tpDistance;
  }

  return {
    stopLoss: Number(stopLoss.toFixed(2)),
    takeProfit: Number(takeProfit.toFixed(2)),
    riskReward: (tpDistance / slDistance).toFixed(2)
  };
}
```
